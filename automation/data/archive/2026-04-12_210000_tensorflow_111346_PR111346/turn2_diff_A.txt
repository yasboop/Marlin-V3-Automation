+bool MsaAlgorithm::MaybeCreateMirroredAllocationForConditionalUse(
+    const AllocationValue& allocation_value, const AllocationValue::Use& use,
+    int64_t use_time, absl::Span<AllocationValue> allocation_values) {
+  const HloUse& hlo_use = use.hlo_use;
+  if (hlo_use.instruction->opcode() != HloOpcode::kConditional) {
+    return false;
+  }
+  if (!options_.enable_conditional_redundant_eviction_elimination) {
+    return false;
+  }
+
+  Allocation* aliased_allocation =
+      GetLiveAllocationAt(*allocation_value.allocation_sequence(), use_time);
+  if (!aliased_allocation ||
+      aliased_allocation->memory_space() != MemorySpace::kAlternate) {
+    return false;
+  }
+
+  const auto& instruction_schedule = hlo_live_range_.instruction_schedule();
+
+  // Check that all uses in all branches can work with alternate memory.
+  for (const AllocationValue::Use& other_use : allocation_value.uses()) {
+    if (other_use.hlo_use.instruction != hlo_use.instruction) {
+      continue;
+    }
+    // Operand 0 of conditional is the predicate, not passed to computations.
+    if (other_use.hlo_use.operand_number == 0) {
+      continue;
+    }
+    HloComputation* called_computation =
+        hlo_use.instruction->called_computations().at(
+            other_use.hlo_use.operand_number - 1);
+    const HloInstruction* parameter_instruction =
+        called_computation->parameter_instruction(0);
+    HloValue* parameter_value =
+        &alias_analysis_.dataflow_analysis().GetUniqueValueAt(
+            parameter_instruction, other_use.hlo_use.operand_index);
+    int64_t parameter_time = instruction_schedule.at(parameter_instruction);
+    int64_t conditional_time = instruction_schedule.at(hlo_use.instruction);
+    int64_t min_use_time = conditional_time;
+    for (const HloUse& parameter_use : parameter_value->GetUses()) {
+      if (parameter_use.instruction->parent() == called_computation &&
+          parameter_use.instruction->opcode() !=
+              HloOpcode::kGetTupleElement &&
+          parameter_use.instruction->opcode() != HloOpcode::kTuple &&
+          parameter_use.instruction->opcode() != HloOpcode::kBitcast) {
+        min_use_time = std::min(
+            min_use_time, instruction_schedule.at(parameter_use.instruction));
+        // Check if this individual use inside the branch requires default
+        // memory. If so, we cannot keep the buffer in alternate memory.
+        if (!options_.is_use_allowed_in_alternate_mem_fn(parameter_use)) {
+          VLOG(4) << "Conditional mirrored allocation not possible for "
+                  << called_computation->name()
+                  << ": use " << parameter_use.ToString()
+                  << " not allowed in alternate memory.";
+          return false;
+        }
+      }
+    }
+    // Check if we can keep the buffer in alternate memory for this branch.
+    if (!options_.prefetch_interval_picker->CanAllocateInAlternateMemoryNoCopy(
+            parameter_value->shape(), parameter_time, min_use_time)) {
+      VLOG(4) << "Conditional mirrored allocation not possible for "
+              << called_computation->name()
+              << ": parameter use too far from parameter definition.";
+      return false;
+    }
+    // Also check if the conditional use itself requires default memory.
+    if (!options_.is_use_allowed_in_alternate_mem_fn(other_use.hlo_use)) {
+      VLOG(4) << "Conditional mirrored allocation not possible: use "
+              << other_use.hlo_use.ToString()
+              << " not allowed in alternate memory.";
+      return false;
+    }
+  }
+
+  // All branches can use alternate memory. Create mirrored allocations for
+  // branch parameters.
+  VLOG(3) << "Creating mirrored allocations in alternate memory for "

+                               GetAliasedOffset(*aliased_allocation));
+  }
+  return true;

+TEST_F(MemorySpaceAssignmentTest,
+       ConditionalSkipsEvictionWhenOperandPinnedInAlternateMem) {
+  // When a buffer is in alternate memory and all uses inside conditional
+  // branches can use alternate memory, the algorithm should skip the eviction
+  // and keep the buffer in alternate memory for the subsequent use.
+  absl::string_view hlo_string = R"hlo(
+  HloModule CondAllocation, is_scheduled=true
+
+  true_computation {
+    p0 = (f32[3]{0}) parameter(0)
+    gte = f32[3]{0} get-tuple-element(p0), index=0
+    ROOT neg1 = f32[3]{0} negate(gte)
+  }
+
+  false_computation {
+    p0 = (f32[3]{0}) parameter(0)
+    gte = f32[3]{0} get-tuple-element(p0), index=0
+    ROOT neg2 = f32[3]{0} negate(gte)
+  }
+
+  ENTRY entry {
+    p0 = f32[3]{0} parameter(0)
+    p1 = pred[] parameter(1)
+    copy = f32[3]{0} copy(p0)
+    tuple = (f32[3]{0}) tuple(copy)
+    conditional = f32[3]{0} conditional(p1, tuple, tuple), true_computation=true_computation, false_computation=false_computation
+    ROOT add = f32[3]{0} add(copy, conditional)
+  }
+  )hlo";
+  TF_ASSERT_OK_AND_ASSIGN(auto module,
+                          ParseAndReturnVerifiedModule(hlo_string));
+  Options options = DefaultMemorySpaceOptions();
+  options.enable_conditional_redundant_eviction_elimination = true;
+  AssignMemorySpace(module.get(), options);

+  // copy should be in alternate memory.

+      module->GetComputationWithName("entry")->GetInstructionWithName("copy");
+  EXPECT_EQ(copy->shape().layout().memory_space(), kAlternateMemorySpace);

+  // Branch parameters should get alternate memory from the mirrored allocation.
